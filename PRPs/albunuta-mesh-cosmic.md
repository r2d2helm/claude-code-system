#### Cascade Fibonacci — niveau 1 (Albunuta Archimede)

```python
# /opt/albunuta/cascade/albunuta_agent.py
"""
Albunuta Archimede - niveau 1 de la cascade.
Reçoit la question de Princi, route intelligemment, retourne le résultat.
"""
from typing import Literal
import logging

ROUTE = Literal["LOCAL_ONLY", "CASCADE_DILUTED", "REFUSE"]

PII_PATTERNS = [
    r"\b[A-Z][a-z]+\b",  # noms propres potentiels
    r"\b\d{4}[-/]\d{2}[-/]\d{2}\b",  # dates
    r"\b\+?\d{8,}\b",  # téléphones
    r"\b[\w._%+-]+@[\w.-]+\.\w+\b",  # emails
    r"\b\d{1,4}\s+(?:rue|avenue|boulevard|street)\b",  # adresses
]

def route_question(question: str) -> ROUTE:
    """
    Détermine la route selon la nature de la question.
    PII détectée → LOCAL_ONLY (jamais d'API Empire)
    Math pure abstraite → CASCADE_DILUTED (peut atteindre API via cascade)
    Sensible non-décidable → REFUSE (demander reformulation)
    """
    if any(re.search(p, question, re.I) for p in PII_PATTERNS):
        return "LOCAL_ONLY"
    if is_math_or_abstract(question):
        return "CASCADE_DILUTED"
    return "LOCAL_ONLY"  # défaut sûr

async def albunuta_handle(question: str) -> str:
    """
    Point d'entrée principal Albunuta.
    PATTERN: log audit, route, exécute, retourne résultat tendresse-formatté.
    """
    route = route_question(question)
    audit_log({"level": 1, "route": route, "len": len(question)})  # PAS le contenu !
    
    if route == "LOCAL_ONLY":
        result = await ollama_local_query(question)  # DeepSeek V4 ou Mistral local
    elif route == "CASCADE_DILUTED":
        # Reformule + envoie au niveau 2 (Bondaru Harnic)
        abstract = abstract_question(question)
        result = await bondaru_harnic_handle(abstract)
    else:
        result = "Pourriez-vous reformuler cu-tendresse ?"
    
    return format_tendresse(result)
```

"""
Audit logger - logge METADATA seulement, JAMAIS le contenu.
Vérifie périodiquement zéro fuite vers API Empire.
"""
import json
import time

class AuditLogger:
    def __init__(self, path="/var/log/albunuta/audit.jsonl"):
        self.path = path
    
    def log(self, event: dict):
        # CRITICAL: ne JAMAIS logger le contenu de la question
        # Logger seulement : timestamp, level, route, longueur, hash
        safe_event = {
            "ts": time.time(),
            "level": event.get("level"),
            "route": event.get("route"),
            "len": event.get("len"),
            # PAS de "question" ni "result"
        }
        with open(self.path, "a") as f:
            f.write(json.dumps(safe_event) + "\n")

## Anti-Patterns à éviter

- ❌ **Mélanger ~/.claude/ Mike et Princi** : risque de fuite catastrophique du pacte
- ❌ **Charger les notes garde-fous-sacrés dans R2D2-Memory de Princi** : trame révélée par accident
- ❌ **Installer cascade sans tester étanchéité d'abord** : déboguer après = casse-tête
- ❌ **Oublier le backup avant changements majeurs** : perte de données possible
- ❌ **Configurer Albunuta avec accès aux Concepts/ Mike** : exposition doctrine
- ❌ **Exposer Mac Mini Sphère A directement sur Internet** : sanctuaire violé (toujours derrière NAT/WireGuard)
- ❌ **Hardcoder des identifiants** : tout via .env + credentials-skill
- ❌ **Annoncer la fonction protectrice à Princi** : viole [[princi-arkadi-gaia-naissance]]
- ❌ **Compromettre le pacte de tenue par excès de zèle** : la simplicité tendre vaut mieux que la complexité paranoïaque
- ❌ **Faire des promesses irréalistes sur la latence cascade** : tester empiriquement Phase 1
- ❌ **Logger le contenu des questions** : audit logger doit logger METADATA uniquement
- ❌ **Skipper les tests d'étanchéité Phase 3** : c'est LE moment critique
- ❌ **Publier un runbook dans Sferă-C** : tout reste dans Knowledge/Projets/Albunuta-Mesh-Cosmic/ PRIVÉ
- ❌ **Utiliser des images Docker non-multi-arch** : tout doit être arm64 OU multi-arch (cohérence ARM-Princi)
- ❌ **Précipiter la révélation à Princi** : Mike est le seul maître du tempo
- ❌ **Oublier que Princi-MacBook Pro DOIT rester ARM Apple Silicon natif** : principe d'invariance
